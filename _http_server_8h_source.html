90  void _readHeader() {

92 
93  auto data(std::make_shared<boost::asio::streambuf>());
94  data->prepare(1024);
95 
96  boost::asio::async_read_until(
97  _socket,
98  *data,
99  "\r\n\r\n",
100  [self, data](const boost::system::error_code &e, size_t n) {
101  if (e) {
102  std::cout << "error in HttpServer::Connection#_readHeader: " << e << std::endl;
103  return;
104  }
105 
106  boost::asio::streambuf::const_buffers_type bufs = data->data();
107  std::string dataAsString(boost::asio::buffers_begin(bufs),
108  boost::asio::buffers_begin(bufs) + n);
109 
110  auto headers(std::make_shared<std::unordered_map<std::string, std::string>>(
111  self->_httpParser.parseHeader(dataAsString, true)));
112 
113  unsigned long long size;
114  auto contentLengthIter = headers->find("Content-Length");
115  if (contentLengthIter != headers->end()) {
116  size = std::stoull(contentLengthIter->second);
117  } else {
118  size = 0;
119  }
120 
121  if (size == 0) {
122  std::string answer = self->_httpParser.generateResponse("Bad request", "text/plain",
123  400, "Bad request", false);
124  boost::asio::async_write(
125  self->_socket,

127  [](const boost::system::error_code &e, size_t n) {});
128  return;
129  }
130 
131  data->consume(n);
